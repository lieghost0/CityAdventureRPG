gameMain = {}
local UITable = require("UI/GameMain/View/gameMainView")

function gameMain:Create()
	local ui = {}
	setmetatable(ui, {__index = UITable})
	ui:Create()
	ui.layer = 1
	UI.LoadUI(ui)
	ui:OnOpen()
	return ui
end

function UITable:OnOpen()
	self:Init()
end

function gameMain.Init()
	Manager.Battle:StartBattle()
end

function UITable:ButtonClickHandler(btn)
	if btn == self.buttonMap then
		gameMain.Init()
	elseif btn == self.buttonQuest then
		-- Manager.UI.OpenUI("loadGame")
	end
end

function Destroy()
	UITable.buttonMap.onClick:RemoveAllListeners()
	UITable.buttonQuest.onClick:RemoveAllListeners()
	UITable.buttonTalent.onClick:RemoveAllListeners()
	UITable.buttonSkill.onClick:RemoveAllListeners()
	UITable.buttonPartner.onClick:RemoveAllListeners()
	UITable.buttonSetting.onClick:RemoveAllListeners()
end
