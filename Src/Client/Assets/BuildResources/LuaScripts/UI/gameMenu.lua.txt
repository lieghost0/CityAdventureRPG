gameMenu = {}
local UITable = require("UI/View/gameMenuView")

function gameMenu:Create()
	local ui = {}
	setmetatable(ui, {__index = UITable})
	ui:Create()
	ui.layer = 1
	UI.LoadUI(ui)
	ui:Init()
	return ui
end



function UITable:ButtonClickHandler(btn)
	if btn == self.buttonNewGame then
		Manager.UI:CloseUI(self.name)
		UI.OpenUI("gameMain")
	elseif btn == self.buttonLoadGame then
		-- Manager.UI.OpenUI("loadGame")
	end
end

function Destroy()
	UITable.buttonNewGame.onClick:RemoveAllListeners()
	UITable.buttonLoadGame.onClick:RemoveAllListeners()
	UITable.buttonSetting.onClick:RemoveAllListeners()
	UITable.buttonExitGame.onClick:RemoveAllListeners()
end
